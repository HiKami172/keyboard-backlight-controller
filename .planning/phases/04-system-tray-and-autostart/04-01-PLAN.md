    Module-level setup (MUST be in this exact order — gi.require_version before any from gi.repository):
    ```python
    import sys, os, json, pathlib
    import gi
    gi.require_version('AyatanaAppIndicator3', '0.1')
    gi.require_version('Gtk', '3.0')
    gi.require_version('GdkPixbuf', '2.0')
    from gi.repository import AyatanaAppIndicator3, Gtk, GdkPixbuf, GLib
    # ProfileManager only uses stdlib (json, pathlib, dataclasses) — safe to import here
    sys.path.insert(0, str(pathlib.Path(__file__).parent.parent.parent))
    from kbd_backlight.profiles.manager import ProfileManager
    ```

    Implement class TrayProcess:

    __init__(self):
    - self._manager = ProfileManager()
    - Create indicator: AyatanaAppIndicator3.Indicator.new('kbd-backlight', 'input-keyboard', AyatanaAppIndicator3.IndicatorCategory.HARDWARE)
    - indicator.set_status(AyatanaAppIndicator3.IndicatorStatus.ACTIVE)
    - Call self._build_menu()
    - Register stdin watcher: GLib.io_add_watch(sys.stdin.fileno(), GLib.IO_IN, self._on_stdin)
    - Store indicator as self._indicator

    _build_menu(self):
    - Create new Gtk.Menu()
    - Call get_all_profiles() on self._manager to get {name: Profile} dict
    - For each (name, profile) in sorted(profiles.items()): append self._make_profile_item(name, profile.r, profile.g, profile.b)
    - If no profiles: append a Gtk.MenuItem(label='(no profiles)') with set_sensitive(False)
    - Append Gtk.SeparatorMenuItem()
    - Append 'Open Settings' Gtk.MenuItem: on activate sends {'action': 'show'}
    - Append 'Quit' Gtk.MenuItem: on activate sends {'action': 'quit'} then calls Gtk.main_quit()
    - CRITICAL: call menu.show_all() BEFORE indicator.set_menu(menu) — AppIndicator spec requirement
    - Store as self._menu (so it isn't garbage collected)

    _make_profile_item(self, name, r, g, b) -> Gtk.MenuItem:
    - Create Gtk.MenuItem()
    - Create Gtk.Box(orientation=Gtk.Orientation.HORIZONTAL, spacing=6)
    - Color swatch: pixbuf = GdkPixbuf.Pixbuf.new(GdkPixbuf.Colorspace.RGB, False, 8, 16, 16)
    - RGBA packing (NOT ARGB): packed = (r << 24) | (g << 16) | (b << 8) | 0xFF
    - pixbuf.fill(packed)
    - For color_cycle profiles where r=g=b=0: use packed = (128 << 24) | (128 << 16) | (128 << 8) | 0xFF (gray swatch)
    - image = Gtk.Image.new_from_pixbuf(pixbuf)
    - label = Gtk.Label(label=name); label.set_xalign(0)
    - box.pack_start(image, False, False, 0); box.pack_start(label, True, True, 0)
    - item.add(box)
    - item.connect('activate', lambda _, n=name: self._on_profile_clicked(n))
    - return item

    _on_profile_clicked(self, name):
    - self._send(json.dumps({'action': 'select_profile', 'name': name}))

    _on_stdin(self, fd, condition):
    - line = sys.stdin.readline().strip()
    - if line == 'REFRESH': self._build_menu()
    - elif line == 'QUIT': Gtk.main_quit(); return False
    - return True  # Keep watching (CRITICAL: return False stops the watch)

    _send(self, message: str):
    - print(message, flush=True)  # flush=True is critical for pipe IPC

    if __name__ == '__main__':
    - TrayProcess()
    - Gtk.main()

    Key pitfalls to avoid:
    - Never import AyatanaAppIndicator3 after GTK4 — this file is GTK3-only
    - menu.show_all() must come BEFORE indicator.set_menu(menu), not after
    - RGBA packing is (r<<24)|(g<<16)|(b<<8)|0xFF, NOT ARGB
    - _on_stdin must return True to keep the GLib watch active; return False only on QUIT
    - flush=True on print() is mandatory — unbuffered stdout for pipe IPC
